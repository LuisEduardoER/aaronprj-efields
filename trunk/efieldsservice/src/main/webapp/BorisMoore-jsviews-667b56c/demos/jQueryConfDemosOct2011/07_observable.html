8 Observable property changes: JsViews

HTML:

...	
	<td data-getfrom="[firstName]" />
	<td>
		<input data-getfrom="[firstName]" data-to="[firstName]" />
	</td>
...

Script:

$( "#details" ).link( person, "#personTmpl" );

// Observable property change
$( "#changeName" ).click( function() {
	$.observable( person ).setProperty( "firstName", person.firstName + "Plus" );
});
